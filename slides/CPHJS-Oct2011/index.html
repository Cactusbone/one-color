Base

one = { // General One.com namespace
    color = function( String | Array | one.color.* | Number ), // Parser
    installColorSpace: function ( str space, str channel+, obj methods )
};
one.installColorSpace("RGB", "Red", "Green", "Blue", "Alpha", {...});
one.color.RGB  = function ( [0;1], [0;1], [0;1], [0;1] )
one.color.HSV  = function ( [0;1], [0;1], [0;1], [0;1] )
one.color.HSL  = function ( [0;1], [0;1], [0;1], [0;1] )
one.color.CMYK = function ( [0;1], [0;1], [0;1], [0;1] )
